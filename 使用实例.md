## 一、curl 命令行示例

### 1.1 认证

```bash
# 登录获取 Token
curl -X POST http://192.168.1.100:8080/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{"username":"admin","password":"admin123"}'

# 响应
{"success":true,"token":"eyJ...","username":"admin"}
```

> 将下面所有 `<TOKEN>` 替换为登录返回的 token

```bash
# 创建数据库

  -H "Content-Type: application/json" \
```

### 完整客户端代码

```python
import json
import requests


class MobileDBClient:
    """移动数据库服务器 REST API 客户端"""

    def __init__(self, host="192.168.1.100", port=8080):
        self.base_url = f"http://{host}:{port}"
        self.token = None
        self.session = requests.Session()

    def login(self, username, password):
        """登录并获取认证令牌"""
        resp = self.session.post(
            f"{self.base_url}/api/auth/login",
            json={"username": username, "password": password}
        )
        data = resp.json()
        if data.get("success"):
            self.token = data["token"]
            self.session.headers.update({"Authorization": f"Bearer {self.token}"})
            print(f"[✓] 登录成功: {data.get('username')}")
            return True
        print(f"[✗] 登录失败: {data.get('error')}")
        return False

    def create_database(self, name):
        """创建数据库"""
        resp = self.session.post(
            f"{self.base_url}/api/databases/create",
            json={"name": name}
        )
        ok = resp.json().get("success", False)
        print(f"[{'✓' if ok else '✗'}] 创建数据库 '{name}'")
        return ok

    def list_databases(self):
        """获取数据库列表"""
        resp = self.session.get(f"{self.base_url}/api/databases")
        dbs = resp.json().get("data", [])
        print(f"\n=== 数据库列表 ({len(dbs)} 个) ===")
        for db in dbs:
            print(f"  - {db['dbName']} ({db.get('displayName', '')})")
        return dbs

    def create_collection(self, db_name, collection_name):
        """创建集合"""
        resp = self.session.post(
            f"{self.base_url}/api/databases/{db_name}/collections/create",
            json={"name": collection_name}
        )
        ok = resp.json().get("success", False)
        print(f"[{'✓' if ok else '✗'}] 创建集合 '{collection_name}'")
        return ok

    def insert_document(self, db_name, collection, key, value):
        """插入文档"""
        resp = self.session.post(
            f"{self.base_url}/api/documents/{db_name}/{collection}",
            json={"key": key, "value": value}
        )
        ok = resp.json().get("success", False)
        print(f"[{'✓' if ok else '✗'}] 插入文档 '{key}'")
        return ok

    def get_document(self, db_name, collection, key):
        """获取单条文档"""
        resp = self.session.get(
            f"{self.base_url}/api/documents/{db_name}/{collection}/{key}"
        )
        result = resp.json()
        if result.get("success") and result.get("data"):
            doc = result["data"][0]
            print(f"\n=== 文档: {key} ===")
            print(json.dumps(doc, indent=2, ensure_ascii=False))
            return doc
        print(f"[✗] 文档不存在")
        return {}

    def query_documents(self, db_name, collection, limit=20, offset=0):
        """查询文档列表"""
        resp = self.session.get(
            f"{self.base_url}/api/documents/{db_name}/{collection}",
            params={"limit": limit, "offset": offset}
        )
        result = resp.json()
        docs = result.get("data", [])
        print(f"\n=== 文档列表 ({len(docs)} 条) ===")
        for doc in docs:
            val = doc.get("value", {})
            print(f"  [{doc.get('doc_key')}] {json.dumps(val, ensure_ascii=False)[:60]}")
        return docs

    def update_document(self, db_name, collection, key, value):
        """更新文档"""
        resp = self.session.put(
            f"{self.base_url}/api/documents/{db_name}/{collection}/{key}",
            json={"value": value}
        )
        ok = resp.json().get("success", False)
        print(f"[{'✓' if ok else '✗'}] 更新文档 '{key}'")
        return ok

    def delete_document(self, db_name, collection, key):
        """删除文档"""
        resp = self.session.delete(
            f"{self.base_url}/api/documents/{db_name}/{collection}/{key}"
        )
        ok = resp.json().get("success", False)
        print(f"[{'✓' if ok else '✗'}] 删除文档 '{key}'")
        return ok

    def change_password(self, old_password, new_password):
        """修改密码"""
        resp = self.session.post(
            f"{self.base_url}/api/users/change-password",
            json={"oldPassword": old_password, "newPassword": new_password}
        )
        data = resp.json()
        if data.get("success"):
            print(f"[✓] 密码修改成功，请重新登录")
        else:
            print(f"[✗] 修改失败: {data.get('error')}")
        return data

    def get_server_status(self):
        """获取服务器状态"""
        resp = self.session.get(f"{self.base_url}/api/admin/status")
        status = resp.json().get("data", {})
        print(f"运行状态: {'运行中' if status.get('isRunning') else '已停止'}")
        print(f"HTTP端口: {status.get('httpPort')}")
        print(f"TCP端口: {status.get('tcpPort')}")
        print(f"活跃连接: {status.get('activeConnections')}")
        return status

    def close(self):
        self.session.close()


# ===== 完整使用示例 =====
def main():
    # 请将 IP 替换为手机的实际 IP 地址
    client = MobileDBClient(host="192.168.1.100")

    # 1. 登录
    print("=" * 40)
    print("1. 登录")
    print("=" * 40)
    if not client.login("admin", "admin123"):
        return

    # 2. 创建数据库和集合
    print("\n" + "=" * 40)
    print("2. 创建数据库和集合")
    print("=" * 40)
    client.create_database("shop")
    client.create_collection("shop", "products")

    # 3. 插入数据
    print("\n" + "=" * 40)
    print("3. 插入商品数据")
    print("=" * 40)
    products = [
        {"key": "p001", "value": {"name": "手机", "price": 2999, "stock": 100}},
        {"key": "p002", "value": {"name": "电脑", "price": 5999, "stock": 50}},
        {"key": "p003", "value": {"name": "耳机", "price": 199, "stock": 200}},
    ]
    for p in products:
        client.insert_document("shop", "products", p["key"], p["value"])

    # 4. 查询列表
    print("\n" + "=" * 40)
    print("4. 查询商品列表")
    print("=" * 40)
    client.query_documents("shop", "products")

    # 5. 查询单条
    print("\n" + "=" * 40)
    print("5. 查询单条商品")
    print("=" * 40)
    client.get_document("shop", "products", "p001")

    # 6. 更新
    print("\n" + "=" * 40)
    print("6. 更新商品价格")
    print("=" * 40)
    client.update_document("shop", "products", "p001",
        {"name": "手机", "price": 2599, "stock": 80})

    # 7. 查看服务器状态
    print("\n" + "=" * 40)
    print("7. 服务器状态")
    print("=" * 40)
    client.get_server_status()

    # 8. 查看数据库列表
    print("\n" + "=" * 40)
    print("8. 数据库列表")
    print("=" * 40)
    client.list_databases()

    client.close()


if __name__ == "__main__":
    main()
```

## 三、JavaScript 完整示例

```javascript
/**
 * MobileDB 客户端 (支持浏览器和 Node.js)
 */
class MobileDBClient {
    constructor(host = '192.168.1.100', port = 8080) {
        this.baseUrl = `http://${host}:${port}`;
        this.token = null;
    }

    async login(username, password) {
        const resp = await fetch(`${this.baseUrl}/api/auth/login`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ username, password })
        });
        const data = await resp.json();
        if (data.success) {
            this.token = data.token;
            console.log(`[✓] 登录成功: ${data.username}`);
            return true;
        }
        console.error(`[✗] 登录失败: ${data.error}`);
        return false;
    }

    async request(method, path, body = null) {
        const headers = { 'Content-Type': 'application/json' };
        if (this.token) headers['Authorization'] = `Bearer ${this.token}`;
        const opts = { method, headers };
        if (body) opts.body = JSON.stringify(body);
        const resp = await fetch(`${this.baseUrl}${path}`, opts);
        return resp.json();
    }

    async listDatabases() {
        return this.request('GET', '/api/databases');
    }

    async createDatabase(name) {
        return this.request('POST', '/api/databases/create', { name });
    }

    async createCollection(dbName, collectionName) {
        return this.request('POST',
            `/api/databases/${dbName}/collections/create`,
            { name: collectionName });
    }

    async insertDocument(dbName, collection, key, value) {
        return this.request('POST',
            `/api/documents/${dbName}/${collection}`,
            { key, value });
    }

    async getDocument(dbName, collection, key) {
        return this.request('GET',
            `/api/documents/${dbName}/${collection}/${key}`);
    }

    async queryDocuments(dbName, collection, limit = 20) {
        return this.request('GET',
            `/api/documents/${dbName}/${collection}?limit=${limit}`);
    }

    async updateDocument(dbName, collection, key, value) {
        return this.request('PUT',
            `/api/documents/${dbName}/${collection}/${key}`,
            { value });
    }

    async deleteDocument(dbName, collection, key) {
        return this.request('DELETE',
            `/api/documents/${dbName}/${collection}/${key}`);
    }

    async changePassword(oldPassword, newPassword) {
        return this.request('POST', '/api/users/change-password',
            { oldPassword, newPassword });
    }

    async getServerStatus() {
        return this.request('GET', '/api/admin/status');
    }
}

// ===== 使用示例 =====
(async () => {
    const client = new MobileDBClient('192.168.1.100');

    // 1. 登录
    if (!await client.login('admin', 'admin123')) return;

    await client.createDatabase('notes_app');
```

## 四、TCP 协议客户端示例

TCP 端口 `9090`，使用自定义二进制协议（4字节长度前缀 + JSON）。

### 消息类型对照

| 类型 | 值 | 方向 | 说明 |
|------|-----|------|------|
| AUTH | 1 | 客户端→服务器 | 认证 |
| INSERT | 3 | 客户端→服务器 | 插入文档 |
| GET | 4 | 客户端→服务器 | 获取文档 |
| QUERY | 5 | 客户端→服务器 | 查询集合 |
| UPDATE | 6 | 客户端→服务器 | 更新文档 |
| DELETE | 7 | 客户端→服务器 | 删除文档 |
| PING | 8 | 客户端→服务器 | 心跳检测 |
| ERROR | 255 | 服务器→客户端 | 错误响应 |

### Python TCP 客户端

```python
import socket
import json
import struct

class TcpClient:
    def __init__(self, host="192.168.1.100", port=9090):
        self.host = host
        self.port = port
        self.sock = None

    def connect(self):
        self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.sock.settimeout(10)
        self.sock.connect((self.host, self.port))
        print(f"[✓] TCP 连接成功 ({self.host}:{self.port})")

    def send_request(self, msg_type, params):
        body = json.dumps({"type": msg_type, "params": params}).encode("utf-8")
        self.sock.sendall(struct.pack("!I", len(body)) + body)
        resp_len = struct.unpack("!I", self.sock.recv(4))[0]
        resp_data = b""
        while len(resp_data) < resp_len:
            resp_data += self.sock.recv(resp_len - len(resp_data))
        return json.loads(resp_data.decode("utf-8"))

    def auth(self, username, password):
        result = self.send_request(1, {"username": username, "password": password})
        print(f"[{'✓' if result.get('success') else '✗'}] TCP 认证")
        return result.get("success", False)

    def insert(self, db, collection, key, value):
        return self.send_request(3, {"db": db, "collection": collection, "key": key, "value": value})

    def get(self, db, collection, key):
        return self.send_request(4, {"db": db, "collection": collection, "key": key})

    def close(self):
        if self.sock:
            self.sock.close()


# ===== 使用示例 =====
client = TcpClient("192.168.1.100", 9090)
client.connect()

if client.auth("admin", "admin123"):
    # 插入
    result = client.insert("mydb", "users", "user1",
        {"name": "张三", "age": 28})
    print("插入结果:", result)

    # 查询
    result = client.get("mydb", "users", "user1")
    print("查询结果:", json.dumps(result, indent=2, ensure_ascii=False))

client.close()
```

### Java TCP 客户端

```java
import java.io.*;
import java.net.*;

public class MobileDBTcpClient {
    private Socket socket;
    private DataInputStream in;
    private DataOutputStream out;

    public void connect(String host, int port) throws IOException {
        socket = new Socket(host, port);
        in = new DataInputStream(socket.getInputStream());
        out = new DataOutputStream(socket.getOutputStream());
    }

    public String sendRequest(int type, String jsonParams) throws IOException {
        String body = String.format("{\"type\":%d,\"params\":%s}", type, jsonParams);
        out.writeInt(body.getBytes().length);
        out.write(body.getBytes());
        out.flush();
        int length = in.readInt();
        byte[] data = new byte[length];
        in.readFully(data);
        return new String(data);
    }

    public String authenticate(String username, String password) throws IOException {
        String params = String.format("{\"username\":\"%s\",\"password\":\"%s\"}", username, password);
        return sendRequest(1, params);
    }

    public void close() throws IOException {
        if (socket != null) socket.close();
    }

    // 使用示例
    public static void main(String[] args) throws Exception {
        MobileDBTcpClient client = new MobileDBTcpClient();
        client.connect("192.168.1.100", 9090);
        String result = client.authenticate("admin", "admin123");
        System.out.println("认证结果: " + result);
        client.close();
    }
}
```

```bash
HOST="192.168.1.100"

echo "=== 1. 健康检查 ==="
curl -s http://$HOST:8080/health | python -m json.tool

echo -e "\n=== 2. 登录获取 Token ==="
LOGIN_RESP=$(curl -s -X POST http://$HOST:8080/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{"username":"admin","password":"admin123"}')
TOKEN=$(echo $LOGIN_RESP | python -c "import sys,json;print(json.load(sys.stdin)['token'])")
echo "Token: ${TOKEN:0:20}..."

echo -e "\n=== 3. 创建数据库 ==="
curl -s -X POST http://$HOST:8080/api/databases/create \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"name":"demo"}'

echo -e "\n=== 4. 创建集合 ==="
curl -s -X POST http://$HOST:8080/api/databases/demo/collections/create \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"name":"users"}'

echo -e "\n=== 5. 批量插入数据 ==="
for i in {1..5}; do
  curl -s -X POST http://$HOST:8080/api/documents/demo/users \
    -H "Authorization: Bearer $TOKEN" \
    -H "Content-Type: application/json" \
    -d "{\"key\":\"user$i\",\"value\":{\"name\":\"用户$i\",\"age\":$((20 + i)),\"score\":$((RANDOM % 100))}}"
  echo ""
done

echo -e "\n=== 6. 查询所有数据 ==="
curl -s "http://$HOST:8080/api/documents/demo/users?limit=10" \
  -H "Authorization: Bearer $TOKEN" | python -m json.tool

echo -e "\n=== 7. 查看服务器状态 ==="
curl -s http://$HOST:8080/api/admin/status \
  -H "Authorization: Bearer $TOKEN" | python -m json.tool
```
